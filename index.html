// Update for fixing ghost effect in ORIG mode to use the same fade ratio calculation as FILL mode function calculateFadeRatio(orig, fill) { const motionThreshold = 0.5; // example threshold value const difference = Math.abs(orig - fill); if (difference < motionThreshold) { return fill * (1 - difference / motionThreshold); } else { return fill; } } // Further logic to apply the blending approach based on the new fade ratio
